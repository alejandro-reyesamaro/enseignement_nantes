// PARFAIT
var n = SaisieEntier("Nombre: ");
var divs = 1;
for (var i = 2; i < n; i++)
  if (n % i == 0)
    divs = divs + i;

if (divs == n)
  Ecrire("Le nombre est parfait");
else
  Ecrire("Le nombre n'est pas parfait");

  }

  Ecrire("Le nombre est premier");
